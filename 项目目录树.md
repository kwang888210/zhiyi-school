```text
zhiyi-campus/
├── backend/
│   ├── pom.xml                                  # Maven 依赖（Spring Boot 3.2 + MyBatis-Plus + JWT + Lombok）
│   └── src/main/
│       ├── java/com/zhiyi/
│       │   ├── ZhiyiApplication.java           # 启动入口
│       │   ├── common/
│       │   │   ├── Result.java                 # 统一返回封装

│       │   │   ├── BusinessException.java      # 业务异常
│       │   │   └── GlobalExceptionHandler.java # 全局异常处理
│       │   ├── config/
│       │   │   ├── WebMvcConfig.java           # CORS + JWT拦截器 + 静态资源
│       │   │   ├── MybatisPlusConfig.java       # 分页插件 + BCrypt Bean
│       │   │   └── MetaObjectHandlerConfig.java # 自动填充 create_at/update_at
│       │   ├── interceptor/
│       │   │   └── JwtInterceptor.java         # JWT 登录拦截器
│       │   ├── utils/
│       │   │   └── JwtUtils.java               # Token 生成/解析/校验
│       │   └── module/
│       │       ├── user/                       # 模块一（A 负责）
│       │       │   ├── entity/SysUser.java
│       │       │   ├── mapper/SysUserMapper.java
│       │       │   ├── service/                # 空目录，待 A 编写
│       │       │   └── controller/             # 空目录，待 A 编写
│       │       ├── item/                       # 模块二（B 负责）
│       │       │   ├── entity/{Category,Item}.java
│       │       │   ├── mapper/{Category,Item}Mapper.java
│       │       │   ├── service/
│       │       │   └── controller/
│       │       ├── social/                     # 模块三（C 负责）
│       │       │   ├── entity/{ChatMessage,ItemFavorite}.java
│       │       │   ├── mapper/{ChatMessage,ItemFavorite}Mapper.java
│       │       │   ├── service/
│       │       │   └── controller/
│       │       ├── trade/                      # 模块四前半（D 负责）
│       │       │   ├── entity/{TradeOrder,WalletLog}.java
│       │       │   ├── mapper/{TradeOrder,WalletLog}Mapper.java
│       │       │   ├── service/
│       │       │   └── controller/
│       │       └── admin/                      # 模块四后半（D 负责）
│       │           ├── entity/{ViolationReport,ViolationLog}.java
│       │           ├── mapper/{ViolationReport,ViolationLog}Mapper.java
│       │           ├── service/
│       │           └── controller/
│       └── resources/
│           ├── application.yml                 # 数据库 + JWT + 文件上传 配置
│           └── mapper/                         # MyBatis XML 目录（复杂 SQL 放这里）
└── frontend/
    └── src/
        ├── api/                                # 接口封装
        ├── router/                             # 路由
        ├── views/                              # 页面（home/login/item/chat/wallet/admin/user）
        ├── components/                         # 公共组件
        ├── utils/                              # 工具函数
        └── assets/                             # 静态资源
```
